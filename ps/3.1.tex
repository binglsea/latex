%!PS
/side 80 def %%定义边长 

/angle { %%定义五角星的一个尖角   
	side neg 0 rlineto
	currentpoint translate
	-36 rotate
	side 0 rlineto } def

/star {
	newpath
	moveto
	angle  %%先定义一个尖角
	4 { currentpoint translate
		108 rotate
		angle } repeat %%然后靠旋转坐车标系后作出其它四个角色
	closepath } def

gsave  %%保存原始坐标系统的信息
300 500 star  %%描出五星的轮廓路径
1 0 0 setrgbcolor  %%填充红色
fill %%绘出红五星,然后轮廓路径就消失
grestore

300 500 star %%重要新闻描出五星的轮廓路径
0 setgray %%加黑边
stroke

showpage
\end{alltt}
